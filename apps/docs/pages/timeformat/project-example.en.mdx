      ```js showLineNumbers copy filename="index.js"
        const { TimeFormat } = require('atosjs');

        const t = new TimeFormat(); // Instance of TimeFormat

        console.log('Starting tests for TimeFormat...\n');

        // Test: Converting to milliseconds
        console.log('Test: convertToMilliseconds');
        console.log(`10s -> ${t.convertToMilliseconds('10s')}ms`);
        console.log(`5m -> ${t.convertToMilliseconds('5m')}ms`);
        console.log(`2h -> ${t.convertToMilliseconds('2h')}ms\n`);

        // Test: Converting from milliseconds
        console.log('Test: convertFromMilliseconds');
        console.log(`3661000ms -> ${t.convertFromMilliseconds(3661000)}\n`);

        // Test: Elapsed time
        console.log('Test: elapsedSince');
        const timestamp = Date.now() - 5000;
        console.log(`Elapsed time (approx. 5000ms) -> ${t.elapsedSince(timestamp)}ms\n`);

        // Test: Repeat with a limited number
        console.log('Test: repeat');
        t.repeat(3, '1s', () => {
        console.log('Repeat callback executed!');
        });

        // Test: Infinite repeat
        console.log('Test: repeatInfinite');
        const infiniteId = t.repeatInfinite('1s', () => {
        console.log('RepeatInfinite callback executed!');
        clearInterval(infiniteId); // Ends after one cycle
        });

        // Test: Countdown
        console.log('Test: countdown');
        t.countdown('2s', () => {
        console.log('Countdown completed!\n');
        });

        // Test: DailyAt
        console.log('Test: dailyAt');
        t.dailyAt(new Date().getHours(), new Date().getMinutes() + 1, () => {
        console.log('Callback dailyAt executed!\n');
        });

        // Test: After
        console.log('Test: after');
        t.after('2s', () => {
        console.log('After callback executed!\n');
        });

        // Test: Every
        console.log('Test: every');
        const everyId = t.every('1s', () => {
        console.log('Every callback executed!');
        clearInterval(everyId); // Ends after one cycle
        });

        console.log('\nTests completed! Check the logs.');

      ```
