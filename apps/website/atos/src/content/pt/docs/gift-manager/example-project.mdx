### Arquivo Principal (`index.js`)

```javascript filename="index.js" showLineNumbers copy
const { GiftManager } = require("atosjs");

(async () => {
  const gift = new GiftManager({
    logging: true, // ativa as mensagens de log (padrÃ£o Ã© 'true')
    atosdb: { filePath: "gifts.sqlite" }, // usando AtosDB com arquivo personalizado
  });

  const coin = await gift.generate({
    type: "coin",
    value: 100,
  });
  
  console.log({
    GiftId: coin,
    log: await gift.view(coin),
  });
})();
```
